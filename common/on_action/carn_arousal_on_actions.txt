# Increase stress every year
yearly_playable_pulse = {
	events = {
		carn_arousal.0001
	}
}

quarterly_playable_pulse = {
	on_actions = {
		start_arousal_tracking_pulse
	}
}

# Check every quarter if arousal tracking should start, since it's only done for the player
# Would be cleaner to catch character switch events directly
start_arousal_tracking_pulse = {
	trigger = {
		is_ai = no
		is_adult = yes
		NOT = { has_character_flag = carn_track_arousal_periodically }
	}
	effect = {		
		add_character_flag = carn_track_arousal_periodically
		trigger_event = {
			id = carn_arousal.0002
			days = 1
		}
	}
}
